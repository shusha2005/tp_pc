# PC Club Booking — команды запуска (VS Code / PowerShell)

## 1) Открой терминал в корне проекта:
# путь должен быть:
# C:\Users\honor\OneDrive\Рабочий стол\компы_тп\pc-club-booking

## 2) Подними PostgreSQL в Docker
docker compose up -d

## 3) Запусти backend (Django)
cd backend
if (-not (Test-Path ".env")) { Copy-Item ".env.example" ".env" }
python -m pip install -r requirements.txt
python manage.py migrate
python manage.py runserver

# Backend будет доступен на:
# http://127.0.0.1:8000

## 4) Запусти frontend во ВТОРОМ терминале VS Code
cd frontend
python -m http.server 5500

# Frontend будет доступен на:
# http://127.0.0.1:5500

## 5) Проверка (опционально, в третьем терминале)
python -c "import urllib.request; print('API', urllib.request.urlopen('http://127.0.0.1:8000/api/').getcode())"
python -c "import urllib.request; print('FRONT', urllib.request.urlopen('http://127.0.0.1:5500/index.html').getcode())"

## 6) Остановка
# Остановить backend/frontend: Ctrl + C в соответствующих терминалах
# Остановить БД:
docker compose down
